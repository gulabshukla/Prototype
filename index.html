          }

Q3. Add a method to calculate sum of all elements in Array in array's protype, use that method to calculate sum for multiple arrays.

  const ParentObj = {
            sum : function(arr) { 
            let sum = 0; 
            for (let i = 0; i < arr.length; i++){
                sum += arr[i]; 
            }
            console.log(sum); 
        }    
        }
        let arr1=[1,2,3,4,5];
        let arr2=[1,2,3,4,5,6];
        let arr3=[1,2,3,4,5,6,7];
        
        ParentObj.sum(arr1); 
        ParentObj.sum(arr2);
        ParentObj.sum(arr3);
    
    
Output: = 15 21 28
